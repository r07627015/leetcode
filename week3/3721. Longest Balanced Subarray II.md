```python
class SegmentTree:
    def __init__(self, n):
        self.n = n
        # 4 * n 是為了保證二元樹節點夠放，避免 index out of range
        self.tree_min = [0] * (4 * n)
        self.tree_max = [0] * (4 * n)
        self.lazy = [0] * (4 * n)

    def _push_up(self, u):
        """用左右子節點的值來更新父節點 u"""
        # 左子節點 index = u * 2, 右子節點 index = u * 2 + 1
        self.tree_min[u] = min(self.tree_min[u * 2], self.tree_min[u * 2 + 1])
        self.tree_max[u] = max(self.tree_max[u * 2], self.tree_max[u * 2 + 1])

    def _apply(self, u, val):
        """將增量 val 套用到節點 u 上"""
        self.tree_min[u] += val
        self.tree_max[u] += val
        self.lazy[u] += val

    def _push_down(self, u):
        """將懶惰標記下傳給左右子節點"""
        if self.lazy[u] != 0:
            self._apply(u * 2, self.lazy[u])
            self._apply(u * 2 + 1, self.lazy[u])
            self.lazy[u] = 0

    def update(self, u, l, r, ql, qr, val):
        """區間更新：將範圍 [ql, qr] 內的值都加上 val"""
        if ql <= l and r <= qr:
            self._apply(u, val)
            return
        
        self._push_down(u)
        mid = (l + r) // 2
        
        # 遞迴更新左右半部
        if ql <= mid:
            self.update(u * 2, l, mid, ql, qr, val)
        if qr > mid:
            self.update(u * 2 + 1, mid + 1, r, ql, qr, val)
            
        self._push_up(u)

    def find_leftmost(self, u, l, r, qr):
        """
        核心邏輯：在線段樹中搜尋最左邊的值為 0 的索引。
        """
        # 1. 剪枝 (Pruning)：如果區間 [l, r] 最小值 > 0 或最大值 < 0，代表沒 0 
        # 或者 l 已經超過我們搜尋的右邊界 qr
        if self.tree_min[u] > 0 or self.tree_max[u] < 0 or l > qr:
            return -1
        
        # 2. 找到葉子節點
        if l == r:
            return l
        
        self._push_down(u)
        mid = (l + r) // 2
        
        # 3. 優先往左子樹找（因為我們要找「最左邊」的平衡起始點）
        res = self.find_leftmost(u * 2, l, mid, qr)
        
        # 4. 如果左邊找不到 (-1)，才去右邊找
        if res == -1:
            res = self.find_leftmost(u * 2 + 1, mid + 1, r, qr)
            
        return res

class Solution:
    def longestBalanced(self, nums: list[int]) -> int:
        n = len(nums)
        # 初始化線段樹 (索引從 1 到 n)
        st = SegmentTree(n + 1)
        ans = 0
        last_pos = {} # 紀錄數字最後出現的位置
        
        for i, x in enumerate(nums, 1):
            # 奇數當 +1，偶數當 -1
            delta = 1 if x % 2 == 1 else -1
            prev = last_pos.get(x, 0)
            
            # 更新區間 [prev + 1, i] 的前綴和
            # 這代表：對於所有起始於 prev+1 到 i 之間的子陣列，它們的和都受 x 的影響
            st.update(1, 1, n, prev + 1, i, delta)
            last_pos[x] = i
            
            # 找到最左邊讓前綴和等於 0 的起始點索引 pos
            pos = st.find_leftmost(1, 1, n, i)
            if pos != -1:
                # 計算子陣列長度：右索引 i - 左索引 pos + 1
                ans = max(ans, i - pos + 1)
                
        return ans
```

- Time Complexity: `O(n log n)`
- Space Complexity: `O(n)`
