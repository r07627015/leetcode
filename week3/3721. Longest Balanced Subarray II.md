```python
class SegmentTree:
    def __init__(self, n):
        self.n = n
        self.tree_min = [0] * (4 * n)
        self.tree_max = [0] * (4 * n)
        self.lazy = [0] * (4 * n)

    def _push_up(self, u):
        self.tree_min[u] = min(self.tree_min[u << 1], self.tree_min[u << 1 | 1])
        self.tree_max[u] = max(self.tree_max[u << 1], self.tree_max[u << 1 | 1])

    def _apply(self, u, val):
        self.tree_min[u] += val
        self.tree_max[u] += val
        self.lazy[u] += val

    def _push_down(self, u):
        if self.lazy[u] != 0:
            self._apply(u << 1, self.lazy[u])
            self._apply(u << 1 | 1, self.lazy[u])
            self.lazy[u] = 0

    def update(self, u, l, r, ql, qr, val):
        if ql <= l and r <= qr:
            self._apply(u, val)
            return
        self._push_down(u)
        mid = (l + r) >> 1
        if ql <= mid: self.update(u << 1, l, mid, ql, qr, val)
        if qr > mid: self.update(u << 1 | 1, mid + 1, r, ql, qr, val)
        self._push_up(u)

    def find_leftmost(self, u, l, r, qr):
        # 核心邏輯：如果區間 [l, r] 的值域不包含 0，直接剪枝
        if self.tree_min[u] > 0 or self.tree_max[u] < 0 or l > qr:
            return -1
        if l == r:
            return l
        self._push_down(u)
        mid = (l + r) >> 1
        # 先找左子樹
        res = self.find_leftmost(u << 1, l, mid, qr)
        if res == -1: # 左邊沒找到 0，才找右邊
            res = self.find_leftmost(u << 1 | 1, mid + 1, r, qr)
        return res

class Solution:
    def longestBalanced(self, nums: list[int]) -> int:
        n = len(nums)
        st = SegmentTree(n + 1)
        ans, last_pos = 0, {}
        
        for i, x in enumerate(nums, 1):
            delta = 1 if x % 2 == 1 else -1
            prev = last_pos.get(x, 0)
            
            # 只更新「第一次出現該數字」對後方左端點的影響
            st.update(1, 1, n, prev + 1, i, delta)
            last_pos[x] = i
            
            # 尋找滿足 sum == 0 的最左起始點
            pos = st.find_leftmost(1, 1, n, i)
            if pos != -1:
                ans = max(ans, i - pos + 1)
        return ans
```

- Time Complexity: `O(n log n)`
- Space Complexity: `O(n)`
