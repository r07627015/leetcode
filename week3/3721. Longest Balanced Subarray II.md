```python
class SegmentTree:
    def __init__(self, size):
        self.size = size
        # 4n 空間確保二元樹節點不越界，存區間極值與懶惰標記
        self.min_val = [0] * (4 * size)
        self.max_val = [0] * (4 * size)
        self.lazy_tag = [0] * (4 * size)

    def update_range(self, left, right, delta):
        """ 對外接口：更新起始點在 [left, right] 範圍內的子陣列總和 """
        self._update(1, 1, self.size, left, right, delta)

    def search_leftmost_zero(self, boundary_right):
        """ 對外接口：在 1 到目前的右邊界中，找最左邊值為 0 的起始點 """
        return self._find(1, 1, self.size, boundary_right)

    def _push_up(self, node):
        """ 向上更新：父節點紀錄左右子樹的極值，用於後續剪枝 """
        self.min_val[node] = min(self.min_val[node * 2], self.min_val[node * 2 + 1])
        self.max_val[node] = max(self.max_val[node * 2], self.max_val[node * 2 + 1])

    def _apply(self, node, val):
        """ 套用更新：修改當前節點數值並累積懶惰標記 """
        self.min_val[node] += val
        self.max_val[node] += val
        self.lazy_tag[node] += val

    def _push_down(self, node):
        """ 下傳標記：將累積的更新傳給子節點，實現 O(log n) 區間操作 """
        if self.lazy_tag[node] != 0:
            self._apply(node * 2, self.lazy_tag[node])
            self._apply(node * 2 + 1, self.lazy_tag[node])
            self.lazy_tag[node] = 0

    def _update(self, node, start, end, q_left, q_right, val):
        # 區間完全包含：直接套用更新並返回
        if q_left <= start and end <= q_right:
            self._apply(node, val)
            return
        
        self._push_down(node)
        mid = (start + end) // 2
        if q_left <= mid:
            self._update(node * 2, start, mid, q_left, q_right, val)
        if q_right > mid:
            self._update(node * 2 + 1, mid + 1, end, q_left, q_right, val)
        self._push_up(node)

    def _find(self, node, start, end, boundary_right):
        # 核心剪枝：如果區間極值不包含 0，或者超出搜尋邊界，直接跳過
        if self.min_val[node] > 0 or self.max_val[node] < 0 or start > boundary_right:
            return -1
        
        # 找到葉子節點且值為 0
        if start == end:
            return start
        
        self._push_down(node)
        mid = (start + end) // 2
        
        # 優先往左找：為了極大化子陣列長度
        res = self._find(node * 2, start, mid, boundary_right)
        if res == -1: # 左邊沒找到 0 才會往右邊找
            res = self._find(node * 2 + 1, mid + 1, end, boundary_right)
        return res

class Solution:
    def longestBalanced(self, nums: list[int]) -> int:
        n = len(nums)
        # 初始化線段樹 (索引從 1 到 n)
        st = SegmentTree(n + 1)
        ans = 0
        last_pos = {} # 紀錄數字最後出現的位置
        
        for i, x in enumerate(nums, 1):
            # 奇數 +1, 偶數 -1。目標：找區間和 = 0
            delta = 1 if x % 2 == 1 else -1
            prev = last_pos.get(x, 0)
            
            # 處理 Distinct：只有第一次出現該數字的起始點區間 [prev+1, i] 需要計分
            st.update_range(prev + 1, i, delta)
            
            # 在線段樹中尋找最左邊的平衡點
            pos = st.search_leftmost_zero(i)
            
            if pos != -1:
                # 計算子陣列長度：右索引 i - 左索引 pos + 1
                ans = max(ans, i - pos + 1)
            
            # 更新此數字最後出現的位置
            last_pos[x] = i
            
        return ans
```

- Time Complexity: `O(n log n)`
- Space Complexity: `O(n)`
