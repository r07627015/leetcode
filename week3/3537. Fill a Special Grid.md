```python
    def specialGrid(self, n: int) -> List[List[int]]:
        size = 1 << n  # 2^n
        grid = [[0] * size for _ in range(size)]
        
        # cur 代表「下一個要填入的數字」
        cur = 0

        def fill(r: int, c: int, length: int) -> None:
            """
            把左上角在 (r, c)，邊長為 length 的子方格填成合法 special grid
            """
            nonlocal cur

            # base case: 1 x 1 直接填值
            if length == 1:
                grid[r][c] = cur
                cur += 1
                return

            half = length // 2

            # 題目要求的象限大小關係：
            # 右上 < 右下 < 左下 < 左上
            #
            # 所以我們照這個順序遞迴填，
            # 先填的會拿到較小數字，後填的會拿到較大數字。

            # 1. 右上 (top-right)
            fill(r, c + half, half)

            # 2. 右下 (bottom-right)
            fill(r + half, c + half, half)

            # 3. 左下 (bottom-left)
            fill(r + half, c, half)

            # 4. 左上 (top-left)
            fill(r, c, half)

        fill(0, 0, size)
        return grid
```

- Time Complexity: `O(4^n)` (`= O(size^2)`, where `size = 2^n`)
- Space Complexity: `O(4^n)` for the grid, plus `O(n)` recursion stack
