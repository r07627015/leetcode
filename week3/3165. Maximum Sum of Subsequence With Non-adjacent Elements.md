
## Python
```python
class Solution:
    def maximumSumSubsequence(self, nums, queries):
        MOD = 10**9 + 7
        NEG = -10**18
        n = len(nums)

        # 每個節點存四個值：
        # [s00, s01, s10, s11]
        # s00 = 左端不選、右端不選
        # s01 = 左端不選、右端選
        # s10 = 左端選、右端不選
        # s11 = 左端選、右端選
        tree = [[NEG, NEG, NEG, NEG] for _ in range(4 * n)]

        def make_leaf(val):
            """
            單一元素區間的四種狀態
            只有兩種合法情況：
            1. 不選這個元素 => s00 = 0
            2. 選這個元素   => s11 = val
            其餘 s01, s10 不可能
            """
            return [0, NEG, NEG, val]

        def merge(left, right):
            """
            合併左右兩段。

            left = [l00, l01, l10, l11]
            right = [r00, r01, r10, r11]

            要避免：
            左段的右端被選 + 右段的左端被選
            """
            l00, l01, l10, l11 = left
            r00, r01, r10, r11 = right

            # 合併後的大區間四種狀態
            res00 = NEG
            res01 = NEG
            res10 = NEG
            res11 = NEG

            # 大區間：左端不選、右端不選
            res00 = max(
                l00 + r00,  # 左右邊界中間沒衝突
                l00 + r10,
                l01 + r00,  # l01 + r10 不行，因為中間相鄰都選
            )

            # 大區間：左端不選、右端選
            res01 = max(
                l00 + r01,
                l00 + r11,
                l01 + r01,  # l01 + r11 不行
            )

            # 大區間：左端選、右端不選
            res10 = max(
                l10 + r00,
                l10 + r10,
                l11 + r00,  # l11 + r10 不行
            )

            # 大區間：左端選、右端選
            res11 = max(
                l10 + r01,
                l10 + r11,
                l11 + r01,  # l11 + r11 不行
            )

            return [res00, res01, res10, res11]

        def build(idx, left, right):
            if left == right:
                tree[idx] = make_leaf(nums[left])
                return

            mid = (left + right) // 2
            build(idx * 2, left, mid)
            build(idx * 2 + 1, mid + 1, right)
            tree[idx] = merge(tree[idx * 2], tree[idx * 2 + 1])

        def update(idx, left, right, pos, val):
            if left == right:
                tree[idx] = make_leaf(val)
                return

            mid = (left + right) // 2
            if pos <= mid:
                update(idx * 2, left, mid, pos, val)
            else:
                update(idx * 2 + 1, mid + 1, right, pos, val)

            tree[idx] = merge(tree[idx * 2], tree[idx * 2 + 1])

        build(1, 0, n - 1)

        ans = 0
        for pos, x in queries:
            update(1, 0, n - 1, pos, x)

            # 根節點代表整個陣列
            best = max(tree[1])

            # 題目允許空子序列
            if best < 0:
                best = 0

            ans = (ans + best) % MOD

        return ans
```

- Time Complexity: `O(n + q log n)`
- Space Complexity: `O(n)`


q 為 query 次數
