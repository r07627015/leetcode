
## Python
```python
class Solution:
    def minimumPairRemoval(self, nums: List[int]) -> int:
        n = len(nums)
        if n <= 1:
            return 0

        # ===== 1. 用陣列模擬雙向鏈結串列 =====
        value = nums[:]                  # 目前每個節點代表的值
        left = [i - 1 for i in range(n)] # 左鄰居
        right = [i + 1 for i in range(n)]# 右鄰居
        right[-1] = -1

        alive = [True] * n              # 節點是否仍存在
        version = [0] * n               # 節點版本號，用來淘汰 heap 舊資料

        # ===== 2. 記錄目前有多少個「相鄰逆序」=====
        # 只要相鄰每一對都滿足 a <= b，整體就是非遞減
        def is_bad(i: int, j: int) -> int:
            if i == -1 or j == -1:
                return 0
            return 1 if value[i] > value[j] else 0

        bad_count = 0
        for i in range(n - 1):
            bad_count += is_bad(i, i + 1)

        # ===== 3. heap 維護「目前可能的最小相鄰 pair」=====
        # 格式：
        # (pair_sum, left_index, right_index, version[left], version[right])
        heap = []

        def add_pair(i: int) -> None:
            """把 i 和它右邊鄰居組成的 pair 放進 heap。"""
            if i == -1 or not alive[i]:
                return

            j = right[i]
            if j == -1 or not alive[j]:
                return

            heapq.heappush(heap, (value[i] + value[j], i, j, version[i], version[j]))

        for i in range(n - 1):
            add_pair(i)

        # ===== 4. 反覆合併，直到 bad_count == 0 =====
        ans = 0

        while bad_count > 0:
            # 找出目前仍有效的最小 pair
            while True:
                pair_sum, i, j, vi, vj = heapq.heappop(heap)

                if (
                    alive[i] and alive[j] and
                    right[i] == j and
                    version[i] == vi and
                    version[j] == vj
                ):
                    break

            # 合併前局部結構：a <-> i <-> j <-> b
            a = left[i]
            b = right[j]

            # 先扣掉舊的逆序貢獻
            bad_count -= is_bad(a, i)
            bad_count -= is_bad(i, j)
            bad_count -= is_bad(j, b)

            # 執行合併：保留 i，刪掉 j
            value[i] += value[j]
            version[i] += 1

            alive[j] = False
            version[j] += 1

            right[i] = b
            if b != -1:
                left[b] = i

            # 加回新的逆序貢獻
            bad_count += is_bad(a, i)
            bad_count += is_bad(i, b)

            # 只有 a 和 i 附近的 pair 可能改變
            add_pair(a)
            add_pair(i)

            ans += 1

        return ans
```

- Time Complexity: `O(n log n)`
- Space Complexity: `O(n)`
