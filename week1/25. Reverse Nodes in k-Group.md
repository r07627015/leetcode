
## Python
```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseKGroup(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        dummy = ListNode(0, head)
        group_prev = dummy

        while True:
            # 1. 找到這組的第 k 個節點
            kth = group_prev
            for _ in range(k):
                kth = kth.next
                if not kth: # 不足 k 直接回傳
                    return dummy.next

            group_next = kth.next

            # 2. 反轉這一組
            prev = group_next
            curr = group_prev.next

            while curr != group_next:
                tmp = curr.next
                curr.next = prev
                prev = curr
                curr = tmp

            # 3. 接回前後鏈結
            tmp = group_prev.next   # 反轉前的頭，反轉後會變尾巴
            group_prev.next = kth
            group_prev = tmp
```

- Time Complexity: `O(n)`
- Space Complexity: `O(1)`
